## path: /etc/nginx/conf.d/my-app.conf

server {
    listen 80;
    listen [::]:80;
    server_name app.example.com;
    # Redirect everything to HTTPS
    location / {
        return 301 https://$host$request_uri;
    }
}
server {
    listen 443 ssl;
    listen [::]:443 ssl;
    http2 on;
    server_name app.example.com;
    ssl_certificate     /etc/ssl/certs/server.crt;
    ssl_certificate_key /etc/ssl/private/server.key;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384;
    ssl_prefer_server_ciphers on;
    client_max_body_size 20m;
    #--- AdGuard Home Dashboard ---#
    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
    }
    #--- DNS-over-HTTPS requests to backend (Unbound) ---#
    location /dns-query {
        grpc_pass grpc://127.0.0.1:4443;
        proxy_ssl_verify off;
        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header Accept            "application/dns-message";
        proxy_set_header Content-Type      "application/dns-message";
        # Desativa buffers para latência mínima
        proxy_buffering off;
        proxy_request_buffering off;
        # Tempo de espera para conexões ociosas (Keep-alive)
        proxy_read_timeout 60s;
        proxy_send_timeout 60s;
    }
