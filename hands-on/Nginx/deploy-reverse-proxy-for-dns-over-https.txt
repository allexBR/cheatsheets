#-----------------------------------------------------------------------#
#  Nginx server block configuration to redirect DNS-over-HTTPS requests #
#-----------------------------------------------------------------------#
tee /etc/nginx/conf.d/doh.conf <<EOF
server {
    listen 80;
    listen [::]:80;
    server_name hostname.domain.tld;
    # Redirect everything to HTTPS
    location / {
        return 301 https://$host$request_uri;
    }
}
server {
    listen 443 ssl;
    listen [::]:443 ssl;
    http2 on;
    server_name hostname.domain.tld;

    access_log  /var/log/nginx/doh.access.log  main;

    ssl_certificate     /etc/ssl/certs/server.crt;
    ssl_certificate_key /etc/ssl/private/server.key;
    #include /etc/nginx/snippets/ssl-params.conf;

    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384;
    ssl_prefer_server_ciphers on;

    client_max_body_size 20m;

    #--- DNS-over-HTTPS ---#
    location = /dns-query {
        grpc_pass          grpc://127.0.0.1:8443;
        proxy_set_header   Host               $host;
        proxy_set_header   X-Real-IP          $remote_addr;
        proxy_set_header   X-Forwarded-For    $proxy_add_x_forwarded_for;
        proxy_set_header   X-Forwarded-Proto  $scheme;
        proxy_ssl_verify   off;
        proxy_redirect     off;
        proxy_buffering    off;
        proxy_request_buffering off;
        proxy_connect_timeout 5s;
        proxy_send_timeout 30s;
        proxy_read_timeout 30s;
        keepalive_timeout  65s;
        limit_except GET POST {
            deny all;
        }
    }
    #--- AdGuard Home ---#
#    location / {
#        proxy_pass http://localhost:3000;

#        proxy_set_header Host              $host;

}
EOF

# Create dhparam which is used to negotiate the perfect forward secrecy policy with clients
openssl dhparam -out /etc/nginx/dhparam.pem 4096

cat /etc/nginx/dhparam.pem

# Create Snippets folder
mkdir -p /etc/nginx/snippets/

# Nginx SSL parameters configuration
tee /etc/nginx/snippets/ssl-params.conf <<EOF
ssl_protocols TLSv1.3;
ssl_prefer_server_ciphers on;
ssl_dhparam /etc/nginx/dhparam.pem;
ssl_ciphers EECDH+AESGCM:EDH+AESGCM;
ssl_ecdh_curve secp384r1;
ssl_session_timeout  10m;
ssl_session_cache shared:SSL:10m;
ssl_session_tickets off;
ssl_stapling on;
ssl_stapling_verify on;
resolver 9.9.9.9 1.0.0.2 valid=300s;
resolver_timeout 5s;
# Disable strict transport security for now. You can uncomment the following
# line if you understand the implications.
#add_header Strict-Transport-Security "max-age=63072000; includeSubDomains; preload";
add_header X-Frame-Options DENY;
add_header X-Content-Type-Options nosniff;
add_header X-XSS-Protection "1; mode=block";
EOF

    worker_connections  1024;
}


http {
    include       /etc/nginx/mime.types;
    default_type  application/octet-stream;

    log_format  main  '$remote_addr - $remote_user [$time_local] "$request" '
                      '$status $body_bytes_sent "$http_referer" '
                      '"$http_user_agent" "$http_x_forwarded_for"';

    access_log  /var/log/nginx/access.log  main;

    sendfile        on;
    #tcp_nopush     on;

    keepalive_timeout  65;

    #gzip  on;

    ## Security Settings
    server_tokens off;
    proxy_hide_header X-Powered-By;
    fastcgi_hide_header X-Powered-By;
    #more_clear_headers Server;

    ## Security Headers
    add_header Content-Security-Policy "block-all-mixed-content; frame-ancestors 'self'; upgrade-insecure-requests";
    add_header Referrer-Policy "no-referrer";
    add_header Strict-Transport-Security "max-age=31536000; includeSubdomains; preload";
    add_header X-Content-Type-Options nosniff;
    add_header X-Frame-Options "SAMEORIGIN";
    add_header X-XSS-Protection "1; mode=block";

    include /etc/nginx/conf.d/*.conf;
}

# HTTP Headers More Filter

# Download the HTTP_HEADERS_MORE_FILTER package and compile it as a dynamic module:
cd /tmp
git clone https://github.com/openresty/headers-more-nginx-module.git

# Download the current source code corresponding to your installed version of NGINX:
wget https://nginx.org/download/nginx-1.28.3.tar.gz
tar zxvf nginx-1.28.3.tar.gz

# Compile the dynamic module and copy it to the Nginx default modules path:
apt install -y apt-utils autoconf automake build-essential libpcre2-dev libpcre3-dev zlib1g-dev libssl-dev libgeoip-dev
cd nginx-1.28.3
./configure --with-compat --add-dynamic-module=../headers-more-nginx-module
make modules
cp objs/ngx_http_headers_more_filter_module.so /etc/nginx/modules
cd ..
